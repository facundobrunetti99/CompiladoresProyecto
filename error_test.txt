int main() {
    int x;
    int y;
    int z;
    bool result;
    
    x = 10;
    y = 5;
    z = 15;
    
    // Prueba de AND
    if (x > 0 && y < 10) {
        result = true;
        z = 100;
    }
    
    // Prueba de OR
    if (x < 0 || y > 0) {
        z = 200;
    }
    
    // Prueba combinada con AND y OR
    if ((x > 5 && y < 10) || z == 15) {
        result = false;
    }
    
    // While con operador lÃ³gico
    while (x > 0 && y < 20) {
        x = x - 1;
        y = y + 1;
    }
    
    return z;
}
